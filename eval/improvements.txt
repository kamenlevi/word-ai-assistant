To insert a numbered list, use the following code:
```javascript
const list = [
  "Plan",
  "Execute",
  "Review"
];
const para = doc.paragraphs.add();
list.forEach(item => {
  const run = para.addRun(item);
  run.font.size = 24;
  para.listItem = {
    level: 0,
    startAt: 1,
    style: "Numbered"
  };
});
await context.sync();
```

To replace text, use the following code:
```javascript
const para = doc.paragraphs.getFirst();
para.text = para.text.replace("Acme", "Globex");
await context.sync();
```

To add a comment, use the following code:
```javascript
const range = doc.body.getRange("$4.2M");
const comment = range.insertComment("check this number");
await context.sync();
```

To add page numbers to the footer, use the following code:
```javascript
const section = doc.sections.getFirst();
const footer = section.headers.first;
const para = footer.paragraphs.add();
para.text = "Page {page}";
para.alignment = "right";
await context.sync();
```

To insert an equation, use the following code:
```javascript
const para = doc.paragraphs.add();
const run = para.addRun();
run.font.subscript = true;
run.text = "E = mc^2";
await context.sync();
```

To count words, use the following code:
```javascript
const text = doc.body.getText();
const wordCount = text.split(" ").length;
return "Your document has " + wordCount + " words.";
```

To list headings, use the following code:
```javascript
const headings = doc.body.getHeadings();
const headingTexts = headings.map(heading => heading.text);
return headingTexts;
```

To insert a block quote, use the following code:
```javascript
const para = doc.paragraphs.add();
para.text = "\"Premature optimization is the root of all evil.\"";
para.leftIndent = 36;
await context.sync();
```

To delete a paragraph, use the following code:
```javascript
const para = doc.paragraphs.getItem(98);
para.delete();
await context.sync();
```

To change the title, use the following code:
```javascript
const title = doc.paragraphs.getFirst();
title.text = "Q4 Results";
